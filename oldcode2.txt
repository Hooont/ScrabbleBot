def textAtDEPRECATED(row, col, vert, move):
    while letterAt(row, col) or letterAtInMove(move, row, col):
        if vert:
            row -= 1
        else:
            col -= 1

    if vert:
        row += 1
    else:
        col += 1
        
    word = ""
    while letterAt(row, col) or letterAtInMove(move, row, col):
        letter = letterAt(row, col);
        if not letter:
            letter = letterAtInMove(move, row, col)
        word += letter
        if vert:
            row += 1
        else:
            col += 1
    return word
